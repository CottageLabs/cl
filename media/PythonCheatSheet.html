Functions

def function_name(arg1, arg2,

                               keyword1=val1, keyword2=val2, ...):

    <function body>

    return return_value

e.g.

def my_function(x, y, z=0):

    sum = x + y + z

    return sum

 

my_function(1, 2) → 3

my_function(1, 2, 3) → 6

my_function(1, 2, y=4) → 7

 

Classes

class ClassName(SuperClass):

    class_variable = static_value

    def __init__(self, value1, <...>):

        self.instance_variable1 = value1

        self.instance_function()

    def instance_function(self, arg1, <...>):

        <function body>

        return return_value

e.g.

class MyClass(object):

    offset = 1

    def __init__(self, value):

        self.value = value

    def get_offset_value(self):

        return MyClass.offset +

                              self.value

 

MyClass.offset → 1

 

c = MyClass(2)

c.value → 2

c.get_offset_value() → 3

Control Flow

if conditional:

    <body>

elif conditional:

    <body>

else:

    <body>

 

 

e.g.

if i == 7:

    print “seven”

elif i == 8:

    print “eight”

else:

    print str(i)

for value in list:

    <body>

    continue

    break

 

e.g.

 

 

for i in [1, 2, 3, 4]:

    if i == 2: continue

    if i == 3: break

    print i

while conditional:

    <body>

    continue

    break

 

e.g.

    print “infinity”

 

Exceptions

try:

    <body>

    raise Exception()

except Exception as e:

    <exception handling>

finally:

    <clean-up>

 

 

e.g.

try:

    database.update()

except Exception as e:

    log.error(e.msg)

    database.abort()

finally:

    database.commit()
